ehof_origin_initialize_system = {
    switch = {
        trigger = galaxy_size
        huge = {
            ehof_origin_initialize_system_2 = {
                ehof_init_min_angle = $ehof_init_min_angle$
                ehof_init_max_angle = $ehof_init_max_angle$
                min_distance = @ehof_init_huge_min_distance
                max_distance = @ehof_init_huge_max_distance

                ehof_init_min_angle = $ehof_init_min_angle$

                max_distance = @ehof_init_large_max_distance
            }
        }
        medium = {
            ehof_origin_initialize_system_2 = {
                ehof_init_min_angle = $ehof_init_min_angle$

        }
        small = {
            ehof_origin_initialize_system_2 = {
                ehof_init_min_angle = $ehof_init_min_angle$
                ehof_init_max_angle = $ehof_init_max_angle$
                min_distance = @ehof_init_small_min_distance

        tiny = {

}

ehof_origin_initialize_system_2 = {
    switch = {
        trigger = has_origin
        origin_ehof_frameworld = {
            no_scope = {
                spawn_system = {
                    min_orientation_angle = $ehof_init_min_angle$
                    max_orientation_angle = $ehof_init_max_angle$
                    min_distance >= $min_distance$
                    max_distance <= $max_distance$
                    hyperlane = no
                    initializer = ehof_frameworld
                    effect = {
                        save_event_target_as = giga_last_ehof_spawn
                    }
                }
            }
        }
        default = {
            no_scope = {
                spawn_system = {
                    min_orientation_angle = $ehof_init_min_angle$
                    max_orientation_angle = $ehof_init_max_angle$
                    min_distance >= $min_distance$
                    max_distance <= $max_distance$
                    hyperlane = no
                    initializer = ehof_default
                    effect = {
                        save_event_target_as = giga_last_ehof_spawn
                    }
                }
            }
        }
    }
}

ehof_origin_generate_start_buildings_and_districts = {

                    has_origin = origin_ehof_shattered_ring

                }
            }
        }
        if = {
            limit = {
                owner = {
                    OR = {
                        has_valid_civic = civic_machine_assimilator
                        has_valid_civic = civic_machine_servitor
                        has_origin = origin_ehof_syncretic_evolution
                        has_origin = origin_ehof_necrophage

            }
            if = {
                limit = {
                    or = {
                        event_target:ehofo_secondary_species = {
                            or = {

                                is_lithoid = yes
                            }
                        }
                        event_target:ehofo_third_species = {
                            or = {
                                has_trait = trait_machine_unit
                                is_lithoid = yes
                            }
                        }
                    }
                }
                owner = {
                    set_country_flag = lithoid_subspecies
                }
            }
            else = {
                owner = {
                    set_country_flag = non_lithoid_subspecies
                }
            }
        }
        if = {
            limit = {
                owner = {
                    has_origin = origin_ehof_tree_of_life
                    is_lithoid_empire = yes
                }
            }
            owner = {
                set_country_flag = non_lithoid_subspecies
            }
        }
        if = {
            limit = { owner = { is_regular_empire = yes } }
            # total - 24 jobs
            # 3 jobs
            if = {
                limit = { NOT = { owner = { has_origin = origin_ehof_payback } } }
                add_district_and_planet_size_if_needed_effect = {
                    district = district_city
                }
            }
            if = {
                limit = {
                    owner = {
                        has_origin = origin_ehof_payback
                        has_valid_civic = civic_agrarian_idyll
                    }
                }
                add_district_and_planet_size_if_needed_effect = {
                    district = district_city
                }
                add_district_and_planet_size_if_needed_effect = {
                    district = district_farming
                }
            }
            if = {
                limit = {
                    owner = { is_guided_sapience_empire = yes }
                }
                add_district_and_planet_size_if_needed_effect = {
                    district = district_farming
                }
            }
            if = {
                limit = {
                    owner = { NOT = { has_valid_civic = civic_agrarian_idyll } }
                }
                while = {
                    count = 2
                    add_district_and_planet_size_if_needed_effect = {
                        district = district_city
                    }
                }
            }
            else = {
                limit = {
                    owner = {
                        is_individual_machine = no
                    }
                }
                while = {
                    count = 2
                    add_district_and_planet_size_if_needed_effect = {
                        district = district_farming
                    }
                }
            }
            if = {
                limit = { owner = { is_megacorp = yes } }
                add_district_and_planet_size_if_needed_effect = {
                    district = district_city
                }
            }
            # 4 - 6 jobs
            if = {
                limit = {
                    owner = {
                        OR = {
                            NOR = {
                                is_lithoid_empire = yes
                                is_individual_machine = yes
                            }
                            has_country_flag = non_lithoid_subspecies
                        }
                    }
                }
                add_district_and_planet_size_if_needed_effect = {
                    district = district_farming
                }
                if = {
                    limit = {
                        owner = {
                            NOR = {
                                has_origin = origin_payback
                                is_individual_machine = yes
                            }
                        }
                    }
                    add_district_and_planet_size_if_needed_effect = {
                        district = district_farming
                    }
